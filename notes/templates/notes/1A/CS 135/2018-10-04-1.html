Design recipe for (listof X)

;; A (listof X) is one of:
;; * empty
;; (cons x (listof X) -> Any)

(define (listof-X-template lox)
    cond [(empty? lox) ...]
             [else (... (first lox)... listof-X-template (rest lox))...])
